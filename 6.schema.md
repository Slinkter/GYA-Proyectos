# SCHEMA — Tipos compartidos Frontend ↔ Backend

## Tipos TypeScript del Frontend (espejo de los DTOs del backend)

```typescript
// shared/types/presupuesto.types.ts

export type EstadoPresupuesto =
  | "BORRADOR"
  | "EMITIDO"
  | "APROBADO"
  | "RECHAZADO"
  | "CANCELADO";

export interface ItemForm {
  id: string;
  producto: string;
  sistema: string;
  descripcion: string;
  altura: number | null;
  ancho: number | null;
  cantidad: number;
  precioUnitario: number;
}

export interface DiagramaForm {
  id: string;
  titulo: string;
  subtitulo: string;
  svgCode: string;
  precio: number | null;
  promptOriginal?: string;
}

export interface CreatePresupuestoDTO {
  obraNombre: string;
  obraDireccion: string;
  tiempoEntrega: string;
  cliente: {
    nombre: string;
    ruc?: string;
    direccion?: string;
  };
  items: Omit<ItemForm, "id">[];
  diagramas: Omit<DiagramaForm, "id">[];
  subtotal: number;
  descuento: number;
  total: number;
  incluyeIgv: boolean;
  notas?: string;
}

export interface PresupuestoResponse {
  id: string;
  numero: string;
  fecha: string;
  estado: EstadoPresupuesto;
  obraNombre: string;
  obraDireccion: string;
  tiempoEntrega: string | null;
  subtotal: number;
  descuento: number;
  total: number;
  incluyeIgv: boolean;
  cliente: { nombre: string; ruc: string | null; direccion: string | null };
  vendedor: { nombre: string; email: string };
  items: ItemForm[];
  diagramas: DiagramaForm[];
  createdAt: string;
}

export interface SvgGenerateRequest {
  descripcion: string;
  // Ejemplo: "Ventana corrediza 2 hojas, ancho 1200mm, alto 900mm, sistema Nova, vidrio 6mm incoloro"
}

export interface SvgGenerateResponse {
  svgCode: string;
  promptUsado: string;
}
```

## Contratos de API

| Endpoint          | Method | Body                            | Response              |
| ----------------- | ------ | ------------------------------- | --------------------- |
| /auth/login       | POST   | { email, password }             | { token, user }       |
| /presupuestos     | GET    | query: { year?, mes?, estado? } | PresupuestoResponse[] |
| /presupuestos/:id | GET    | —                               | PresupuestoResponse   |
| /presupuestos     | POST   | CreatePresupuestoDTO            | PresupuestoResponse   |
| /presupuestos/:id | PATCH  | Partial<CreatePresupuestoDTO>   | PresupuestoResponse   |
| /svg/generate     | POST   | SvgGenerateRequest              | SvgGenerateResponse   |
