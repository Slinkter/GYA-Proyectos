## SCHEMA PRISMA (fuente de verdad)

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id           String        @id @default(uuid())
  email        String        @unique
  password     String
  nombre       String
  rol          Rol           @default(VENDEDOR)
  activo       Boolean       @default(true)
  createdAt    DateTime      @default(now())
  presupuestos Presupuesto[]
}

enum Rol {
  ADMIN
  VENDEDOR
}

model Cliente {
  id           String        @id @default(uuid())
  nombre       String
  ruc          String?
  direccion    String?
  createdAt    DateTime      @default(now())
  presupuestos Presupuesto[]
}

model Presupuesto {
  id             String      @id @default(uuid())
  numero         String      @unique  // "03-000020" generado automáticamente
  serie          String      @default("03")
  correlativo    Int
  fecha          DateTime    @default(now())
  fechaEntrega   DateTime?
  estado         Estado      @default(BORRADOR)
  obraNombre     String
  obraDireccion  String
  tiempoEntrega  String?     // "3-4 días hábiles"
  subtotal       Decimal     @db.Decimal(10, 2)
  descuento      Decimal     @default(0) @db.Decimal(10, 2)
  total          Decimal     @db.Decimal(10, 2)
  incluyeIgv     Boolean     @default(true)
  notas          String?
  cliente        Cliente     @relation(fields: [clienteId], references: [id])
  clienteId      String
  vendedor       User        @relation(fields: [vendedorId], references: [id])
  vendedorId     String
  items          Item[]
  diagramas      Diagrama[]
  createdAt      DateTime    @default(now())
  updatedAt      DateTime    @updatedAt
}

enum Estado {
  BORRADOR
  EMITIDO
  APROBADO
  RECHAZADO
  CANCELADO
}

model Item {
  id             String      @id @default(uuid())
  orden          Int         // posición en la tabla
  producto       String      // "VENTANA", "MAMPARA", "PUERTA"
  sistema        String?     // "Nova", "Serie 25", etc.
  descripcion    String
  altura         Decimal?    @db.Decimal(8, 2)  // en cm
  ancho          Decimal?    @db.Decimal(8, 2)   // en cm
  cantidad       Int         @default(1)
  precioUnitario Decimal     @db.Decimal(10, 2)
  subtotal       Decimal     @db.Decimal(10, 2)
  presupuesto    Presupuesto @relation(fields: [presupuestoId], references: [id], onDelete: Cascade)
  presupuestoId  String
}

model Diagrama {
  id            String      @id @default(uuid())
  orden         Int
  titulo        String      // "Mampara", "Ventana – Sala"
  subtitulo     String?     // "Incoloro 8 mm"
  svgCode       String      @db.Text  // SVG completo generado por Gemini
  precio        Decimal?    @db.Decimal(10, 2)
  prompt        String?     @db.Text  // Prompt original que generó el SVG
  presupuesto   Presupuesto @relation(fields: [presupuestoId], references: [id], onDelete: Cascade)
  presupuestoId String
}
```

## REGLAS DE CÓDIGO

### NestJS General

- Cada módulo es autocontenido: module + controller + service + dto
- SIEMPRE usar @UseGuards(JwtAuthGuard) en rutas protegidas
- SIEMPRE usar DTOs con class-validator para validar entrada
- SIEMPRE usar @ApiTags y @ApiOperation (Swagger)
- Response format consistente:
  { data: T, message: string, statusCode: number }

### Auth

- Login: POST /auth/login → { email, password } → { token, user }
- Token JWT con expiración 24h
- Guard aplica a TODOS los endpoints excepto /auth/login y /health
- En Fase 1: usuarios creados directamente en DB con seed
- Preparado para Firebase en Fase 3 (estrategia intercambiable)

### Presupuestos — Lógica de numeración automática

```typescript
// En presupuestos.service.ts
async getNextCorrelativo(serie: string): Promise<number> {
  const ultimo = await this.prisma.presupuesto.findFirst({
    where: { serie },
    orderBy: { correlativo: 'desc' },
  })
  return (ultimo?.correlativo ?? 0) + 1
}

formatNumero(serie: string, correlativo: number): string {
  return `${serie}-${String(correlativo).padStart(6, '0')}`
}
```

### Endpoints CRUD Presupuestos

- GET /presupuestos → Lista con filtros (year, mes, estado)
- GET /presupuestos/:id → Detalle completo con items y diagramas
- POST /presupuestos → Crear (genera número automáticamente)
- PATCH /presupuestos/:id → Actualizar
- DELETE /presupuestos/:id → Solo ADMIN, solo BORRADOR
- PATCH /presupuestos/:id/estado → Cambiar estado (BORRADOR→EMITIDO→etc)
